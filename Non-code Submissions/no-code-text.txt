#include <stdio.h>
int main(){
    int num1, num2, product;
    
    // Asking for input
    printf("Enter the first number: ");
    scanf("%d", &num1);
    printf("Enter the second number: ");
    scanf("%d", &num2);
    
    // Multiply two numbers
    product = num1 * num2;
    
    // Displaying result
    printf("Product of %d and %d is: %d", num1, num2, product);
    
    return 0;
}
